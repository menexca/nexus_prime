Ejecutar Primero.

Este script no solo crea las tablas, sino que implementa la integridad referencial, lógica de auditoría automática mediante funciones y disparadores (triggers), e índices de rendimiento para garantizar una respuesta inmediata en tus 60 estaciones de trabajo.

-- 1. EXTENSIONES Y LIMPIEZA (Opcional)
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 2. TABLA DE USUARIOS DEL SISTEMA (Gestión Interna)
CREATE TABLE IF NOT EXISTS usuarios_sistema (
    id_usuario SERIAL PRIMARY KEY,
    usuario_login VARCHAR(20) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    nombre_completo VARCHAR(100),
    estatus BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. TABLA MAESTRO DE PROVEEDORES (Multi-Empresa y Optimizada)
CREATE TABLE IF NOT EXISTS maestro_proveedores (
    -- Contexto y Clave Primaria Compuesta
    cod_compania INT NOT NULL,
    cod_proveedor VARCHAR(20) NOT NULL,
    
    -- Identidad
    rif VARCHAR(15) NOT NULL,
    nit VARCHAR(15),
    nombre_provider VARCHAR(150) NOT NULL,
    
    -- Ubicación y Contactos
    direccion TEXT,
    direccion_alt TEXT,
    zona VARCHAR(50),
    contacto1 VARCHAR(100),
    contacto2 VARCHAR(100),
    email1 VARCHAR(100),
    email2 VARCHAR(100),
    telefono1 VARCHAR(50),
    telefono2 VARCHAR(50),
    fax VARCHAR(50),
    
    -- Información Tributaria
    tipo_persona VARCHAR(20),      -- Natural/Jurídica
    cod_retencion VARCHAR(10),     -- Código ISLR
    sujeto_ret_iva BOOLEAN DEFAULT FALSE,
    tipo_contrib VARCHAR(30),      -- Especial/Ordinario
    contribuyente_iva BOOLEAN DEFAULT TRUE,
    figura_tributaria VARCHAR(50),
    registro_unico VARCHAR(50),
    
    -- Información Contable y Negocio
    forma_pago VARCHAR(30),
    gastos VARCHAR(25),            -- Cuenta contable
    anticipos VARCHAR(25),         -- Cuenta contable
    cxp_proveedor VARCHAR(25),      -- Cuenta contable
    lineas TEXT,
    num_cuenta_ban VARCHAR(25),
    beneficiario VARCHAR(150),
    nombre_banco VARCHAR(50),
    
    -- Estado y Notas
    activo_inactivo BOOLEAN DEFAULT TRUE,
    comentario TEXT,
    
    -- Auditoría (Rastreo Interno)
    id_user_crea INT NOT NULL,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_user_mod INT,
    fecha_modifica TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    -- Restricciones de Integridad
    PRIMARY KEY (cod_compania, cod_proveedor),
    CONSTRAINT fk_user_crea FOREIGN KEY (id_user_crea) REFERENCES usuarios_sistema(id_usuario),
    CONSTRAINT fk_user_mod FOREIGN KEY (id_user_mod) REFERENCES usuarios_sistema(id_usuario),
    CONSTRAINT uk_compania_rif UNIQUE (cod_compania, rif)
);

-- 4. ÍNDICES DE RENDIMIENTO
-- Optimiza búsquedas por RIF y nombre dentro de cada empresa
CREATE INDEX idx_prov_rif ON maestro_proveedores (cod_compania, rif);
CREATE INDEX idx_prov_nombre ON maestro_proveedores (cod_compania, nombre_provider);

-- 5. FUNCIÓN DE AUDITORÍA AUTOMÁTICA
-- Actualiza la fecha de modificación automáticamente al hacer un UPDATE
CREATE OR REPLACE FUNCTION fn_actualizar_fecha_mod()
RETURNS TRIGGER AS $$
BEGIN
    NEW.fecha_modifica = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 6. TRIGGER DE AUDITORÍA
CREATE TRIGGER trg_audit_proveedores
BEFORE UPDATE ON maestro_proveedores
FOR EACH ROW
EXECUTE FUNCTION fn_actualizar_fecha_mod();
