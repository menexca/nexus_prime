CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Función general para actualizar la fecha de modificación automáticamente [cite: 10, 16, 21, 36]
CREATE OR REPLACE FUNCTION fn_actualizar_fecha_mod()
RETURNS TRIGGER AS $$
BEGIN
    NEW.fecha_modifica = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 1.1 Tabla de Usuarios (Corregida a seg_usuarios)
CREATE TABLE IF NOT EXISTS seg_usuarios (
    id_usuario SERIAL PRIMARY KEY,
    usuario_login VARCHAR(20) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    nombre_completo VARCHAR(100),
    email VARCHAR(100), -- Agregado para soportar el campo del formulario de PyQt
    rol VARCHAR(20) DEFAULT 'Operador',
    estatus BOOLEAN DEFAULT TRUE,
    
    -- Auditoría interna (Self-referencing)
    id_user_crea INT,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_user_mod INT,
    fecha_modifica TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Agregar Foreign Keys de auditoría a seg_usuarios (Evita errores al crear la tabla en blanco) [cite: 14, 15]
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.table_constraints WHERE constraint_name = 'fk_usr_crea') THEN
        ALTER TABLE seg_usuarios ADD CONSTRAINT fk_usr_crea FOREIGN KEY (id_user_crea) REFERENCES seg_usuarios(id_usuario);
    END IF;
    IF NOT EXISTS (SELECT 1 FROM information_schema.table_constraints WHERE constraint_name = 'fk_usr_mod') THEN
        ALTER TABLE seg_usuarios ADD CONSTRAINT fk_usr_mod FOREIGN KEY (id_user_mod) REFERENCES seg_usuarios(id_usuario);
    END IF;
END $$;

-- Trigger Auditoría Usuarios
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'trg_audit_usuarios') THEN
        CREATE TRIGGER trg_audit_usuarios
        BEFORE UPDATE ON seg_usuarios
        FOR EACH ROW
        EXECUTE FUNCTION fn_actualizar_fecha_mod();
    END IF;
END $$;

-- 2.1 Catálogo Fiscal CIIU [cite: 25]
CREATE TABLE IF NOT EXISTS fiscal_ciiu (
    id_ciiu SERIAL PRIMARY KEY,
    codigo_ciiu VARCHAR(10) UNIQUE NOT NULL,
    descripcion TEXT NOT NULL,
    ramo_actividad VARCHAR(100)
);

-- 2.2 Tabla Configuración Empresas (Multi-Tenant) - Corregida a cfg_empresas [cite: 26, 27]
CREATE TABLE IF NOT EXISTS cfg_empresas (
    cod_compania SERIAL PRIMARY KEY,
    
    -- Identificación
    razon_social VARCHAR(200) NOT NULL,
    rif VARCHAR(20) NOT NULL UNIQUE,
    nit VARCHAR(20),
    
    -- Ubicación
    pais VARCHAR(50) DEFAULT 'Venezuela',
    estado VARCHAR(50),
    municipio VARCHAR(50),
    ciudad VARCHAR(50),
    zona_postal VARCHAR(10),
    direccion TEXT,
    
    -- Tributario
    tipo_contribuyente VARCHAR(50),
    tipo_contrib_iva VARCHAR(50),
    persona_fiscal VARCHAR(100),
    cod_contribuyente VARCHAR(50),
    porcentaje_patente NUMERIC(5,2),
    id_ciiu_principal INT, 
    
    -- Contacto
    telefono1 VARCHAR(20),
    telefono2 VARCHAR(20),
    telefono3 VARCHAR(20),
    
    -- Estado y Auditoría
    estatus BOOLEAN DEFAULT TRUE,
    id_user_crea INT,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_user_mod INT,
    fecha_modifica TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_ciiu FOREIGN KEY (id_ciiu_principal) REFERENCES fiscal_ciiu(id_ciiu),
    CONSTRAINT fk_empresa_usr_crea FOREIGN KEY (id_user_crea) REFERENCES seg_usuarios(id_usuario),
    CONSTRAINT fk_empresa_usr_mod FOREIGN KEY (id_user_mod) REFERENCES seg_usuarios(id_usuario)
);

-- Trigger Auditoría Empresas [cite: 21]
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'trg_audit_empresas') THEN
        CREATE TRIGGER trg_audit_empresas
        BEFORE UPDATE ON cfg_empresas
        FOR EACH ROW
        EXECUTE FUNCTION fn_actualizar_fecha_mod();
    END IF;
END $$;

-- 2.3 Seguridad: Acceso Usuario-Empresa [cite: 1]
CREATE TABLE IF NOT EXISTS sys_acceso_empresas (
    id_usuario INT NOT NULL,
    cod_compania INT NOT NULL,
    PRIMARY KEY (id_usuario, cod_compania),
    CONSTRAINT fk_acceso_user FOREIGN KEY (id_usuario) REFERENCES seg_usuarios(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_acceso_emp FOREIGN KEY (cod_compania) REFERENCES cfg_empresas(cod_compania) ON DELETE CASCADE

    contribuyente_iva BOOLEAN DEFAULT TRUE,
    figura_tributaria VARCHAR(50),
    registro_unico VARCHAR(50),

    gastos VARCHAR(25),
    anticipos VARCHAR(25),
    cxp_proveedor VARCHAR(25),
    lineas TEXT,
    num_cuenta_ban VARCHAR(25),

-- MÓDULO 4: PRODUCTOS / INVENTARIO
-- =========================================================================
CREATE TABLE IF NOT EXISTS maestro_productos (
    cod_compania INT NOT NULL,
    num_item SERIAL,
    cod_producto VARCHAR(25) NOT NULL,
    
    -- Datos Básicos [cite: 30]
    nombre_producto VARCHAR(200) NOT NULL,
    descripcion TEXT,
    origen VARCHAR(20) DEFAULT 'NACIONAL',
    unidad VARCHAR(20) DEFAULT 'UNIDAD',
    
    -- Ubicación [cite: 31]
    almacen VARCHAR(50),
    pasillo VARCHAR(20),
    estante VARCHAR(20),
    peldano VARCHAR(20),
    
    -- Inventario
    cantidad NUMERIC(12,3) DEFAULT 0,
    pendiente NUMERIC(12,3) DEFAULT 0,
    devuelto NUMERIC(12,3) DEFAULT 0,
    documento_dev VARCHAR(50),
    fecha_elaboracion DATE,
    fecha_vencimiento DATE,
    
    -- Estructura de Costos y Precios [cite: 32]
    costo_unitario NUMERIC(16,2) DEFAULT 0,
    porc_descuento NUMERIC(5,2) DEFAULT 0,
    monto_descuento NUMERIC(16,2) DEFAULT 0,
    iva_porc NUMERIC(5,2) DEFAULT 16.00,
    monto_iva NUMERIC(16,2) DEFAULT 0,
    neto NUMERIC(16,2) DEFAULT 0,
    
    -- Contabilidad [cite: 33]
    tipo_costo VARCHAR(50),
    id_proveedor VARCHAR(20),
    cuenta_por_pagar VARCHAR(25),
    
    -- Estado y Auditoría
    comentario TEXT,
    estatus BOOLEAN DEFAULT TRUE,
    id_user_crea INT,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_user_mod INT,
    fecha_modifica TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (cod_compania, cod_producto),
    CONSTRAINT fk_prod_compania FOREIGN KEY (cod_compania) REFERENCES cfg_empresas(cod_compania),
    CONSTRAINT fk_prod_user_crea FOREIGN KEY (id_user_crea) REFERENCES seg_usuarios(id_usuario),
    CONSTRAINT fk_prod_user_mod FOREIGN KEY (id_user_mod) REFERENCES seg_usuarios(id_usuario)
);

-- Índice de Búsqueda [cite: 35]
CREATE INDEX IF NOT EXISTS idx_prod_nombre ON maestro_productos (cod_compania, nombre_producto);
